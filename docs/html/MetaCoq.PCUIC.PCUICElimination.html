Lemma leq_universe_prop (Σ : global_env_ext) u1 u2 :
  @check_univs cf = true →
  
  wf Σ →
  @leq_universe cf (global_ext_constraints Σ) u1 u2 →

  (is_prop_sort u1 ∧ is_prop_sort u2).
